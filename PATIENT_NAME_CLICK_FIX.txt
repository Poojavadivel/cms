═══════════════════════════════════════════════════════════════════════════════
  JSX CHANGES
═══════════════════════════════════════════════════════════════════════════════

Before:
<div className="patient-name clickable" onClick={onViewPatient}>
  {patientName}
</div>

After:
<div 
  className="patient-name clickable" 
  onClick={(e) => {
    e.stopPropagation();  /* Prevents event bubbling */
    onViewPatient();       /* Opens profile dialog */
  }}
>
  {patientName}
</div>
